%
% The definitions can be placed anywhere in the document body
% and their order is sorted by <symbol> automatically when
% calling makeindex in the makefile
%
% The \glossary command has the following syntax:
%
% \glossary{entry}
%
% The \nomenclature command has the following syntax:
%
% \nomenclature[<prefix>]{<symbol>}{<description>}
%
% where <prefix> is used for fine tuning the sort order,
% <symbol> is the symbol to be described, and <description> is
% the actual description.

% ----------------------------------------------------------------------
% Roman symbols [r]
\nomenclature[rC]{$n_{e}$}{Electron density.}
\nomenclature[rC]{$f_{pe}$}{Electron plasma frequency.}
\nomenclature[rC]{$f_{ce}$}{Electron cyclotron frequency.}
\nomenclature[rC]{$N$}{Refraction index.}

% ----------------------------------------------------------------------
% Greek symbols [g]
\nomenclature[g]{$\tau_{g}$}{Group delay.}
\nomenclature[g]{$\beta$}{Ratio of the plasma pressure to the magnetic pressure.}

\nomenclature[s]{$co$}{At the cutoff.}

\nomenclature[s]{$xpt$}{At the x-point.}
\nomenclature[s]{$sep$}{At the separatrix.}
\nomenclature[s]{$t$}{Referring to the target.}
\nomenclature[s]{$u$}{Referring to upstream.}
